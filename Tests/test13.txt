#begin
int main() {
    arr A[5];
    for (int i = 0; i < 5; i++;) {
        A[i] = 2 * i;
    }

    ptr p;
    p = malloc(5);
    for (int i = 0; i < 5; i++;) {
        p[i] = 3 * i;
    }

    arr B[5];
    for (int i = 0; i < 5; i++;) {
        B[i] = A[i] + p[i];
    }
    print(B); // "[0, 5, 10, 15, 20]"
    free(p);
}
#end
